  <script>
    let comments = JSON.parse(localStorage.getItem('comments') || '[]');
    const commentsList = document.getElementById('comments-list');
    const commentForm = document.getElementById('comment-form');
    const commentUsername = document.getElementById('comment-username');
    const commentText = document.getElementById('comment-text');

    function renderComments() {
      if (comments.length === 0) {
        commentsList.innerHTML = 'No hay comentarios aún. ¡Sé el primero!';
        return;
      }
      commentsList.innerHTML = '';
      comments.forEach(c => {
        const p = document.createElement('p');
        p.style.borderBottom = '1px solid #2a3550';
        p.style.paddingBottom = '8px';
        p.style.marginBottom = '10px';
        p.innerHTML = '<strong>' + escapeHtml(c.user) + ':</strong> ' + escapeHtml(c.text);
        commentsList.appendChild(p);
      });
    }

    function escapeHtml(text) {
      return text.replace(/&/g, "&amp;")
                 .replace(/</g, "&lt;")
                 .replace(/>/g, "&gt;")
                 .replace(/"/g, "&quot;")
                 .replace(/'/g, "&#039;");
    }

    commentForm.addEventListener('submit', e => {
      e.preventDefault();
      const user = commentUsername.value.trim();
      const text = commentText.value.trim();
      if (user.length < 3) {
        alert('El nombre debe tener al menos 3 caracteres.');
        return;
      }
      if (text.length < 3) {
        alert('El comentario debe tener al menos 3 caracteres.');
        return;
      }
      comments.push({ user, text });
      localStorage.setItem('comments', JSON.stringify(comments));
      commentUsername.value = '';
      commentText.value = '';
      renderComments();
    });

    renderComments();
  </script>
